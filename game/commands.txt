     xpos 0.7 #Расположение на экране
     linear 7.0 xpos 0.6 #Линейное движение (Время, ось, куда)

    "Мне нравится": #Вариант 1
        er "Да, мне нравится, а тебе?"

    if ymiko_love == 0:
        e "Я тебя не люблю"

        e "Я тебя люблю"
